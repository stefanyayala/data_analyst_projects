CREATE VIEW v_clientes AS
SELECT 
c.*, 
g.DESCRIPCION,
DATEDIFF(YEAR, FECHA_NACIMIENTO, GETDATE()) AS EDAD

FROM dbo.clientes$ c
LEFT JOIN dbo.genero$ g ON c.GENERO = g.COD_GENERO;
GO

CREATE VIEW v_saldos_productos AS
SELECT TOP (5) p.*, 
m.DESCRIPCION,
m.COTIZACION,
p.SALDO_CAPITAL + p.SALDO_INTERES AS SALDO_TOTAL

FROM dbo.productos$ p
LEFT JOIN dbo.moneda$ m ON p.MONEDA = m.COD_MONEDA
GO

SELECT * FROM dbo.v_saldos_productos

SELECT *,
SALDO_TOTAL * COTIZACION AS SALDO_GS
FROM
dbo.v_saldos_productos

SELECT TIPO_CARTERA, 
SUM(SALDO_TOTAL * COTIZACION) AS SALDO_GS

FROM
dbo.v_saldos_productos
GROUP BY
TIPO_CARTERA
